@app.route('/api/unaki/quick-add-client', methods=['POST'])
def quick_add_client():
    try:
        data = request.get_json()
        
        # Validate required fields
        required_fields = ['first_name', 'last_name', 'phone', 'gender']
        for field in required_fields:
            if not data.get(field):
                return jsonify({'success': False, 'error': f'Missing required field: {field}'}), 400
        
        # Check if client already exists
        existing_client = Client.query.filter_by(phone=data['phone']).first()
        if existing_client:
            return jsonify({
                'success': False, 
                'error': 'A client with this phone number already exists'
            }), 400
        
        # Create new client
        new_client = Client(
            first_name=data['first_name'],
            last_name=data['last_name'],
            phone=data['phone'],
            email=data.get('email', ''),
            gender=data['gender']
        )
        
        db.session.add(new_client)
        db.session.commit()
        
        return jsonify({
            'success': True,
            'message': 'Client added successfully',
            'client': {
                'id': new_client.id,
                'first_name': new_client.first_name,
                'last_name': new_client.last_name,
                'phone': new_client.phone,
                'email': new_client.email
            }
        })
    
    except Exception as e:
        db.session.rollback()
        print(f"Error adding client: {str(e)}")
        return jsonify({'success': False, 'error': str(e)}), 500
