function setupScrollSync() {
  const panel = document.querySelector('.timeline-panel');
  const staff = document.querySelector('.staff-sidebar');
  const header = document.querySelector('.timeline-header');

  if (!panel || !staff || !header) return;

  // flags to avoid feedback loop
  let syncingFromPanel = false;
  let syncingFromStaff = false;

  // When timeline scrolls: sync vertical to staff, and keep header aligned horizontally
  panel.addEventListener('scroll', () => {
    // vertical sync to staff sidebar
    if (!syncingFromStaff) {
      syncingFromPanel = true;
      staff.scrollTop = panel.scrollTop;
    }
    syncingFromStaff = false;

    // Keep sticky header aligned with horizontal scroll (works across browsers)
    // If your header already scrolls naturally with the content, you can comment this line.
    header.style.marginLeft = (-panel.scrollLeft) + 'px';
  }, { passive: true });

  // When staff list scrolls: sync vertical back to timeline
  staff.addEventListener('scroll', () => {
    if (!syncingFromPanel) {
      syncingFromStaff = true;
      panel.scrollTop = staff.scrollTop;
    }
    syncingFromPanel = false;
  }, { passive: true });
}
