<script>
function createQuickPrepaid(type) {
    if (confirm(`Create all standard ${type} packages?`)) {
        const formData = new FormData();
        formData.append('prepaid_type', type);
        
        fetch('/packages/prepaid/create-quick', {
            method: 'POST',
            body: formData
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                alert(data.message);
                location.reload();
            } else {
                alert('Error: ' + data.message);
            }
        });
    }
}

function assignPackage(packageId, packageName) {
    document.getElementById('assign_package_id').value = packageId;
    document.querySelector('#assignPackageModal .modal-title').textContent = `Assign ${packageName}`;
    new bootstrap.Modal(document.getElementById('assignPackageModal')).show();
}

function assignServicePackage(packageId, packageName) {
    assignPackage(packageId, packageName);
}

function submitAssignment() {
    const form = document.getElementById('assignPackageForm');
    const formData = new FormData(form);
    const packageId = document.getElementById('assign_package_id').value;
    
    fetch(`/packages/prepaid/assign/${packageId}`, {
        method: 'POST',
        body: formData
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            alert(data.message);
            bootstrap.Modal.getInstance(document.getElementById('assignPackageModal')).hide();
            location.reload();
        } else {
            alert('Error: ' + data.message);
        }
    });
}
</script>
