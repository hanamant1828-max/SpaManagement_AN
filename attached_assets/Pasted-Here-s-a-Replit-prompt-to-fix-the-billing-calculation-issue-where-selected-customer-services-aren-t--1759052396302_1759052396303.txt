```javascript
function calculateBillingTotal() {
    console.log('Calculating billing total...');
    let subtotal = 0;
    
    // Get all selected services
    const selectedServices = document.querySelectorAll('.selected-service');
    selectedServices.forEach(service => {
        const amount = parseFloat(service.dataset.amount || 0);
        console.log('Adding service amount:', amount);
        subtotal += amount;
    });
    
    console.log('Subtotal before tax:', subtotal);
    updateBillingDisplay(subtotal);
}
```

**3. Fix Service Addition to Bill**
Ensure the "Add to Bill" button properly adds services:

```javascript
function addToBill(serviceElement) {
    serviceElement.classList.add('selected-service');
    const amount = serviceElement.querySelector('.amount').textContent;
    serviceElement.dataset.amount = amount.replace('₹', '');
    calculateBillingTotal();
}
```

**4. Update Tax Calculation Display**
Fix the final billing section to show correct amounts:
```javascript
function updateBillingDisplay(subtotal) {
    document.getElementById('subtotalAmount').textContent = `₹${subtotal.toFixed(2)}`;
    
    const cgst = subtotal * 0.09;
    const sgst = subtotal * 0.09;
    const total = subtotal + cgst + sgst;
    
    document.getElementById('totalAmount').textContent = `₹${total.toFixed(2)}`;
}
```
